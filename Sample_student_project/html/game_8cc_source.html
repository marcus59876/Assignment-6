80 int game::eval_with_lookahead(int look_ahead, int beat_this)
81 // Evaluate a board position with lookahead.
82 // --int look_aheads: How deep the lookahead should go to evaluate the move.
83 // --int beat_this: Value of another move that we're considering. If the
84 // current board position can't beat this, then cut it short.
85 // The return value is large if the position is good for the player who just
86 // moved.
87 {
88  queue<string> moves; // All possible opponent moves
89  int value; // Value of a board position after opponent moves
90  int best_value; // Evaluation of best opponent move
91  game* future; // Pointer to a future version of this game
92 
93  // Base case:
94  if (look_ahead == 0 || is_game_over( ))
95  {
96  if (last_mover( ) == COMPUTER)
97  return evaluate( );
98  else
99  return -evaluate( );
100  }
101 
102  // Recursive case:
103  // The level is above 0, so try all possible opponent moves. Keep the
104  // value of the best of these moves from the opponent's perspective.
105  compute_moves(moves);
106  // assert(!moves.empty( ));
107  best_value = INT_MIN;
108  while (!moves.empty( ))
109  {
110  future = clone( );
111  future->make_move(moves.front( ));
112  value = future->eval_with_lookahead(look_ahead - 1, best_value);
113  delete future;
114  if (value > best_value)
115  {
116  if (-value <= beat_this)
117  return INT_MIN + 1; // Alpha-beta pruning
118  best_value = value;
119  }
120  moves.pop( );
121  }
122 
123  // The value was calculated from the opponent's perspective.
124  // The answer we return should be from player's perspective, so multiply times -1:
125  return -best_value;
126 }
